docker version версия докер клиента и daemona
docker ps -a список запущенных и остановлеенных контейнеров | опция -a нужна для показа остановленных контейнеров
docker images список всех образов на компьютере
docker run название_образа создание контейнера |-it название образа интерактивный терминал| -d запуск в фоновом режиме|--name присвоить контейнеру имя| -p публикация порта внешнйи порт 8080:80 порт контейнера 
docker pull название_образа скачать из репозитория этот образ
docker rm id_контейнера_или_имя_контейнера удалить контейнер
docker container prune удалить все остановленные контейнеры
docker container inspect id_контейнера подробные детали контейнера
docker stop/kill id_контейнера или имя_контейнера
docker exec опции id_контейнера_или_имя_контейнера название процесса 
docker run -v ${pwd}(путь_к_локальной_папке_на_пк):/usr/share/nginx/html(путь_к_папке_внутри_контейнера) название образа передать в контейнер папку с компьютера
docker run --rm название_образа  автоматическое удаление контейнера при его остановке
\ в конце строки позволяет разделить команду на много строк
